31fn symlink_dir<S: AsRef<Path>, D: AsRef<Path>>(src: S, dest: D) -> anyhow::Result<()> {
32    let src = src.as_ref();
33    let dest = dest.as_ref();
34
35    cfg_if::cfg_if! {
36        if #[cfg(windows)] {
37            use std::os::windows::fs::symlink_dir;
38            symlink_dir(src, dest)
39                .with_context(|| format!("failed to create symlink from \"{}\" to \"{}\"", dest.display(), src.display()))?;
40
41            Ok(())
42        } else if #[cfg(unix)] {
43            use std::os::unix::fs::symlink;
44            symlink(src, dest)
45                .with_context(|| format!("failed to create symlink from \"{}\" to \"{}\"", dest.display(), src.display()))?;
46
47            Ok(())
48        } else {
49            bail!("symlinks are not supported on this platform");
50        }
51    }
52}

121    /// Build the debian sysroot.
122    pub fn build(&mut self) -> anyhow::Result<DebianSysroot> {
123        let path = self.base_path.join(&self.release).join(&self.arch);
124
125        // TODO: Consider deferring, in the case that the provided release/arch is invalid.
126        let sysroot = path.join("sysroot");
127        std::fs::create_dir_all(&path)?;
128        std::fs::create_dir_all(&sysroot)?;
129        std::fs::create_dir_all(path.join("packages"))?;
130
131        let lock_path = path.join("lock");
132        let mut lock = fslock::LockFile::open(&lock_path)?;
133        lock.lock()?;
134
135        let database = rusqlite::Connection::open(path.join("database.db"))?;
136        database.execute_batch(SETUP_SQL)?;
137
138        // Setup usrmerge if on trixie.
139        //
140        // The usrmerge package doesn't work on Windows due to needing perl,
141        // which needs ":" in paths.
142        if self.release.as_str() == "trixie" {
143            let usr = sysroot.join("usr");
144            let usr_lib = usr.join("lib");
145            let usr_lib64 = usr.join("lib64");
146
147            let lib = sysroot.join("lib");
148            let lib64 = sysroot.join("lib64");
149
150            std::fs::create_dir_all(&usr)?;
151
152            if !lib.try_exists()? {
153                symlink_dir(usr_lib, lib)?;
154            }
155
156            if !lib64.try_exists()? {
157                symlink_dir(usr_lib64, lib64)?;
158            }
159        }
160
161        let http = ureq::agent();
162        Ok(DebianSysroot {
163            path,
164            lock,
165
166            database,
167            http,
168
169            repository_url: self.repository_url.clone(),
170            release: self.release.clone(),
171            arch: self.arch.clone(),
172
173            pending_install: HashSet::new(),
174        })
175    }
176}

194impl DebianSysroot {
195    /// Refresh the on-disk package list.
196    pub fn update(&mut self) -> anyhow::Result<()> {
197        let package_list_path = self.path.join("Packages.txt");
198
199        if package_list_path.try_exists()? {
200            return Ok(());
201        }
202
203        let packages_url = format!(
204            "{}/dists/{}/main/binary-{}/Packages.gz",
205            self.repository_url, self.release, self.arch
206        );
207
208        let mut response = self.http.get(&packages_url).call()?;
209        ensure!(response.status() == 200);
210
211        let response_reader = response.body_mut().as_reader();
212        let mut response_reader = GzDecoder::new(response_reader);
213        let mut contents = String::new();
214        response_reader.read_to_string(&mut contents)?;
215
216        let transaction = self.database.transaction()?;
217        {
218            let mut statement = transaction.prepare_cached(
219                "
220                INSERT OR REPLACE INTO packages (
221                    name,
222                    file_name,
223                    depends
224                ) VALUES (
225                    :name,
226                    :file_name,
227                    :depends
228                );
229                ",
230            )?;
231            for package in parse_package_list(&contents) {
232                let package = package?;
233
234                let depends = if package.depends.is_empty() {
235                    None
236                } else {
237                    Some(package.depends.join(", "))
238                };
239
240                statement.execute(named_params! {
241                    ":name": package.name,
242                    ":file_name": package.file_name,
243                    ":depends": depends,
244                })?;
245            }
246        }
247
248        transaction.commit()?;
249
250        File::create(package_list_path)?;
251
252        Ok(())
253    }

255    /// Get the info for a given package by name.
256    fn get_package_info(&mut self, name: &str) -> anyhow::Result<Option<PackageInfo>> {
257        let mut statement = self.database.prepare_cached(
258            "
259            SELECT
260                name,
261                file_name, 
262                depends
263            FROM
264                packages
265            where
266                name = :name;
267        ",
268        )?;
269        statement
270            .query_row(
271                named_params! {
272                    ":name": name,
273                },
274                |row| {
275                    let name = row.get("name")?;
276                    let file_name = row.get("file_name")?;
277                    let depends: Option<String> = row.get("depends")?;
278
279                    let depends = match depends {
280                        Some(depends) => depends.split(", ").map(Into::into).collect(),
281                        None => Vec::new(),
282                    };
283
284                    Ok(anyhow::Ok(PackageInfo {
285                        name,
286                        file_name,
287                        depends,
288                    }))
289                },
290            )
291            .optional()?
292            .transpose()
293    }

332    /// Install a package, if needed.
333    pub fn install(&mut self, install_package_name: &str) -> anyhow::Result<()> {
334        if self
335            .get_installed_package_info(install_package_name)?
336            .is_some()
337        {
338            return Ok(());
339        }
340
341        if !self.pending_install.insert(install_package_name.into()) {
342            return Ok(());
343        }
344
345        self.update()?;
346
347        let package = self
348            .get_package_info(install_package_name)?
349            .with_context(|| format!("missing package \"{install_package_name}\""))?;
350
351        for dep in package.depends.iter() {
352            let package = dep.split_once(' ').map(|(name, _rest)| name).unwrap_or(dep);
353            self.install(package)?;
354        }
355
356        let downloaded_package_path = self
357            .path
358            .join("packages")
359            .join(format!("{install_package_name}.deb"));
360        if !downloaded_package_path.try_exists()? {
361            let deb_url = format!("{}/{}", self.repository_url, package.file_name);
362            let mut response = self.http.get(&deb_url).call()?;
363            ensure!(response.status() == 200);
364
365            let tmp_file_path = self
366                .path
367                .join("packages")
368                .join(format!("{install_package_name}.deb.tmp"));
369
370            let mut tmp_file = File::create(&tmp_file_path)?;
371            std::io::copy(&mut response.body_mut().as_reader(), &mut tmp_file)?;
372            tmp_file.flush()?;
373            tmp_file.sync_all()?;
374
375            std::fs::rename(&tmp_file_path, &downloaded_package_path)?;
376        }
377
378        let mut deb = ar::Archive::new(File::open(downloaded_package_path)?);
379        let mut count = 0;
380        while let Some(entry) = deb.next_entry() {
381            let entry = entry?;
382            let header = entry.header();
383            let identifier = std::str::from_utf8(header.identifier())?;
384
385            if count == 2 {
386                let (_rest, extension) = identifier
387                    .rsplit_once('.')
388                    .context("expected an extension")?;
389
390                let mut tar = match extension {
391                    "xz" => {
392                        let reader = XzDecoder::new(entry);
393                        tar::Archive::new(reader)
394                    }
395                    _ => {
396                        bail!("unknown extension \"{extension}\"");
397                    }
398                };
399
400                // TODO: Do this is a way thats safer, esp during failure?
401                let path = self.path.join("sysroot");
402                for entry in tar.entries()? {
403                    let mut entry = entry?;
404                    let header = entry.header();
405
406                    // This is a hack for windows.
407                    if header.entry_type().is_symlink() {
408                        let src = entry.link_name()?.context("missing link name")?;
409                        let dst = path.join(entry.path()?);
410
411                        let src = match src.strip_prefix("/") {
412                            Ok(src) => src.to_owned(),
413                            Err(_err) => dst.parent().context("dst missing parent")?.join(src),
414                        };
415                        let src = path.join(src);
416
417                        if let Some(parent) = dst.parent() {
418                            std::fs::create_dir_all(parent)?;
419                        }
420
421                        let src_metadata = std::fs::metadata(&src).with_context(|| {
422                            format!("failed to get metadata for file at {}", src.display())
423                        })?;
424
425                        // TODO: Copy Dir
426                        if src_metadata.is_dir() {
427                            continue;
428                        }
429
430                        std::fs::copy(&src, &dst).with_context(|| {
431                            format!(
432                                "failed to symlink-copy \"{}\" to \"{}\"",
433                                src.display(),
434                                dst.display()
435                            )
436                        })?;
437
438                        continue;
439                    }
440
441                    entry.unpack_in(&path)?;
442                }
443
444                break;
445            }
446
447            count += 1;
448        }
449
450        self.database
451            .prepare_cached(
452                "
453                    INSERT OR REPLACE INTO installed_packages (
454                        name,
455                        file_name,
456                        depends
457                    ) VALUES (
458                        :name, 
459                        :file_name,
460                        :depends
461                    );
462                
463                ",
464            )?
465            .execute(named_params! {
466                ":name": package.name,
467                ":file_name": package.file_name,
468                ":depends": package.depends.join(", "),
469            })?;
470
471        Ok(())
472    }

491/// Package info
492#[derive(Debug)]
493pub struct PackageInfo {
494    /// The name of the package
495    pub name: String,
496
497    /// The package file name
498    pub file_name: String,
499
500    /// Package dependencies
501    pub depends: Vec<String>,
502}
503
504/// Parse a package list.
505pub fn parse_package_list(input: &str) -> impl Iterator<Item = anyhow::Result<PackageInfo>> + '_ {
506    input.trim_end().split("\n\n").map(|package| {
507        let mut name = None;
508        let mut file_name = None;
509        let mut depends = None;
510        for line in package.split('\n') {
511            // TODO: Multiline support
512            let (key, value) = match line.split_once(": ") {
513                Some(value) => value,
514                None => {
515                    continue;
516                }
517            };
518
519            match key {
520                "Package" => {
521                    ensure!(name.is_none());
522                    name = Some(value);
523                }
524                "Filename" => {
525                    ensure!(file_name.is_none());
526                    file_name = Some(value);
527                }
528                "Depends" => {
529                    ensure!(depends.is_none());
530                    depends = Some(value);
531                }
532                _ => {}
533            }
534        }
535        let name = name.context("missing package name")?.into();
536        let file_name = file_name.context("missing package file name")?.into();
537        let depends = depends
538            .map(|depends| depends.split(", ").map(Into::into).collect())
539            .unwrap_or(Vec::new());
540
541        Ok(PackageInfo {
542            name,
543            file_name,
544            depends,
545        })
546    })
