8/// Get the debian arch for a given triple
9pub fn get_debian_arch(triple: &str) -> Option<&'static str> {
10    match triple {
11        "x86_64-unknown-linux-gnu" => Some("amd64"),
12        "aarch64-unknown-linux-gnu" => Some("arm64"),
13        "armv7-unknown-linux-gnueabihf" => Some("armhf"),
14        _ => None,
15    }
16}
17
18/// Get the gcc triple for a given Rust triple
19pub fn get_gcc_triple(triple: &str) -> Option<&'static str> {
20    match triple {
21        "x86_64-unknown-linux-gnu" => Some("x86_64-linux-gnu"),
22        "aarch64-unknown-linux-gnu" => Some("aarch64-linux-gnu"),
23        _ => None,
24    }
25}

27#[derive(Debug, argh::FromArgs)]
28#[argh(description = "a tool to build Rust projects using a Debian sysroot")]
29struct Options {
30    #[argh(option, description = "the Rust target triple")]
31    target: String,
32
33    #[argh(
34        option,
35        description = "the package to build",
36        short = 'p',
37        long = "package"
38    )]
39    package: Option<String>,
40
41    #[argh(
42        option,
43        description = "the debian package to install",
44        long = "install-package"
45    )]
46    install_package: Vec<String>,
47
48    #[argh(option, description = "build features")]
49    features: Option<String>,
50
51    #[argh(option, description = "the target debian release")]
52    debian_release: Option<String>,
53}
54
55fn main() -> anyhow::Result<()> {
56    let options: Options = argh::from_env();
57
58    let metadata = MetadataCommand::new().exec()?;
59
60    let target_dir = metadata.workspace_root.join("target");
61
62    let target = options.target.as_str();
63    let package = options.package.as_deref();
64
65    let debian_arch = get_debian_arch(target)
66        .with_context(|| format!("failed to get debian arch for \"{target}\""))?;
67    let gcc_triple = get_gcc_triple(target)
68        .with_context(|| format!("failed to get gcc triple for \"{target}\""))?;
69
70    let sysroot_path = target_dir.join("debian-sysroot");
71    let mut sysroot_builder = DebianSysrootBuilder::new(sysroot_path.into());
72    sysroot_builder.arch(debian_arch);
73    if let Some(debian_release) = options.debian_release.as_ref() {
74        sysroot_builder.release(debian_release);
75    }
76    let mut sysroot = sysroot_builder.build()?;
77
78    for package in options.install_package.iter() {
79        let package = package.replace("%DEBIAN_ARCH%", debian_arch);
80        println!("Installing {package}");
81
82        sysroot.install(&package)?;
83    }
84
85    let sysroot = sysroot.get_sysroot_path();
86    let sysroot = sysroot.to_str().context("sysroot path is not unicode")?;
87    // openssl chokes on backslashes in paths
88    let sysroot = sysroot.replace('\\', "/");
89
90    let cc = "clang";
91    let cflags = format!("--sysroot={sysroot} -fuse-ld=lld");
92    let rustflags =
93        format!("-Clinker=clang -Clink-args=--target={target} -Clink-args=--sysroot={sysroot} -Clink-args=-fuse-ld=lld");
94    let pkg_config_libdir = format!("{sysroot}/usr/lib/{gcc_triple}/pkgconfig");
95
96    let mut command = Command::new("cargo");
97    command.env("CC", cc);
98    command.env("CFLAGS", cflags);
99    command.env("RUSTFLAGS", rustflags);
100
101    // This will prevent the default paths pkg-config paths from being used.
102    // Test this with libssl-dev.
103    command.env("PKG_CONFIG_DIR", "");
104    command.env("PKG_CONFIG_SYSROOT_DIR", sysroot);
105    command.env("PKG_CONFIG_LIBDIR", pkg_config_libdir);
106
107    command.arg("build");
108    command.arg("--release");
109    if let Some(package) = package {
110        command.args(["-p", package]);
111    }
112    command.args(["--target", target]);
113    if let Some(features) = options.features {
114        command.args(["--features", features.as_str()]);
115    }
116    let output = command.status().context("failed to spawn cargo build")?;
117    ensure!(output.success(), "failed to run cargo build");
118
119    Ok(())
120}
