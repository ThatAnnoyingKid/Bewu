10/*
11Something less full-blown than a full actor system, more modular.
12Task -> TaskSystem.
13
14Tasks spawn on the task system.
15The task system will join all tasks and handle errors that may occur.
16The Task struct is given a message to process.
17
18Task handles should be typed by message.
19TaskHandle<UserMessage>.
20These handles can then be passed as the constructor for tasks.
21let taskhandle = TaskSystem.spawn(TaskHandle);
22let newtaskobj = UserTask::new(taskhandle);
23let handle = TaskSystem.spawn(newtaskobj);
24
25type safety.
26
27How to bringdown?
28reverse order? internal graph? task cycles?
29Idea: Punt to user.
30Send shutdown system message to all.
31Programmer must close a critical task at that point, so shutdown can occur.
32Handle shutdown by just allowing to close tasks via handles? await the system?
33system should abort all on drop. losing the manager is not acceptable.
34*/

36/// Library error type
37#[derive(Debug, thiserror::Error)]
38pub enum Error {
39    #[error("the task is closed")]
40    TaskClosed,
41
42    #[error("the task did not reply to the message")]
43    NoResponse,
44
45    #[error("the task is missing its internal handle")]
46    MissingHandle,
47
48    #[error(transparent)]
49    TokioJoin(#[from] tokio::task::JoinError),
50}
51
52/// Internal message enum of all messages a task can accept.
53enum TaskMessage<M> {
54    /// A close message
55    Close {
56        tx: tokio::sync::oneshot::Sender<()>,
57    },
58
59    /// A user-defined message
60    User(M),
61}
62
63/// Spawn a task with the given channel capacity.
64pub fn spawn<T>(mut task: T, capacity: usize) -> TaskHandle<T::Message>
65where
66    T: Task,
67{
68    let (tx, mut rx) = tokio::sync::mpsc::channel(capacity);
69    let (weak_task_handle_tx, weak_task_handle_rx) = tokio::sync::oneshot::channel();
70
71    let handle = tokio::spawn(async move {
72        let weak_task_handle = match weak_task_handle_rx.await {
73            Ok(weak_task_handle) => weak_task_handle,
74            Err(_error) => {
75                return;
76            }
77        };
78
79        let mut context = TaskContext::new(weak_task_handle);
80
81        loop {
82            tokio::select! {
83                Some(result) = context.join_set.join_next() => {
84                    task.process_join_result(result, &mut context);
85                }
86                message = rx.recv() => {
87                    match message {
88                        Some(TaskMessage::Close { tx }) => {
89                            rx.close();
90                            task.process_close(&mut context);
91                            let _ = tx.send(()).is_ok();
92                        }
93                        Some(TaskMessage::User(message)) => {
94                            task.process_message(message, &mut context);
95                        }
96                        None => {
97                            break;
98                        }
99                    }
100                }
101            }
102        }
103
104        while let Some(result) = context.join_set.join_next().await {
105            task.process_join_result(result, &mut context);
106        }
107    });
108
109    let task_handle = TaskHandle::new(tx, handle);
110
111    // This should never fail.
112    // We make it an option to avoid requiring Debug for Message.
113    // This may be fixed with a custom Debug impl for WeakTaskHandle and/or TaskHandle.
114    weak_task_handle_tx
115        .send(task_handle.downgrade())
116        .ok()
117        .unwrap();
118
119    task_handle
120}
