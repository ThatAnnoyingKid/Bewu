9/// A master playlist
10#[derive(Debug)]
11pub struct MasterPlaylist {
12    /// A list of all variant streams
13    pub variant_streams: Vec<VariantStream>,
14}
15
16impl std::str::FromStr for MasterPlaylist {
17    type Err = Error;
18
19    fn from_str(input: &str) -> Result<Self, Self::Err> {
20        let mut lines = input.lines();
21
22        let start_tag = lines.next().ok_or(Error::UnexpectedEof)?;
23        if start_tag != EXT_M3U_TAG {
24            return Err(Error::InvalidStartTag {
25                tag: start_tag.into(),
26            });
27        }
28
29        let mut stream_info = None;
30        let mut variant_streams = Vec::with_capacity(4);
31        for line in lines {
32            if line.is_empty() {
33                continue;
34            }
35
36            if let Some(line) = line.strip_prefix('#') {
37                if line.starts_with("EXT") {
38                    let tag: Tag = line.parse::<Tag>()?;
39
40                    match tag {
41                        Tag::ExtXStreamInf {
42                            bandwidth,
43                            average_bandwidth,
44                            codecs,
45                            resolution,
46                            frame_rate,
47                            video_range,
48                            name,
49                        } => {
50                            if stream_info.is_some() {
51                                return Err(Error::DuplicateTag {
52                                    tag: EXT_X_STREAM_INF_TAG,
53                                });
54                            }
55
56                            // TODO: Ensure this is immediately followed by a uri somehow.
57                            stream_info = Some((
58                                bandwidth,
59                                average_bandwidth,
60                                codecs,
61                                resolution,
62                                frame_rate,
63                                video_range,
64                                name,
65                            ));
66                        }
67                        _ => {
68                            return Err(Error::InvalidTag);
69                        }
70                    }
71                }
72            } else {
73                let uri = UriReferenceStr::new(line).map_err(|error| Error::InvalidUri {
74                    line: line.into(),
75                    error,
76                })?;
77                let (
78                    bandwidth,
79                    average_bandwidth,
80                    codecs,
81                    resolution,
82                    frame_rate,
83                    video_range,
84                    name,
85                ) = stream_info.take().ok_or(Error::MissingTag {
86                    tag: EXT_X_STREAM_INF_TAG,
87                })?;
88
89                variant_streams.push(VariantStream {
90                    uri: uri.into(),
91                    bandwidth,
92                    average_bandwidth,
93                    codecs,
94                    resolution,
95                    frame_rate,
96                    video_range,
97                    name,
98                });
99            }
100        }
101
102        Ok(Self { variant_streams })
103    }
104}
