15fn main() -> anyhow::Result<()> {
16    let config = Config::load_path("config.toml").context("failed to load config")?;
17    crate::logger::init(&config).context("failed to init logger")?;
18
19    let tokio_rt = tokio::runtime::Builder::new_multi_thread()
20        .enable_all()
21        .build()?;
22
23    tokio_rt.block_on(async_main(config))
24}

26async fn async_main(config: Config) -> anyhow::Result<()> {
27    let app_state = Arc::new(AppState::new(&config.data_directory).await?);
28    let app = self::routes::routes(&config, app_state.clone())?;
29    let server_listener = tokio::net::TcpListener::bind(&config.bind_address)
30        .await
31        .with_context(|| format!("failed to bind to address \"{}\"", config.bind_address))?;
32
33    info!("listening on \"{}\"", config.bind_address);
34
35    let (shutdown_tx, mut shutdown_rx) = tokio::sync::oneshot::channel();
36    let mut server_task_handle = tokio::spawn(async move {
37        axum::serve(server_listener, app)
38            .with_graceful_shutdown(async {
39                let result = tokio::signal::ctrl_c()
40                    .await
41                    .context("failed to register ctrl+c handler");
42                let _ = shutdown_tx.send(()).is_ok();
43                match result {
44                    Ok(()) => {
45                        info!("shutting down");
46                    }
47                    Err(error) => {
48                        error!("{error:?}");
49                    }
50                }
51            })
52            .await
53            .context("server error")
54    });
55
56    let server_result = tokio::select! {
57        result = &mut server_task_handle => Some(result),
58        _ = &mut shutdown_rx => None,
59    };
60
61    let server_result = match server_result {
62        Some(result) => result,
63        None => {
64            let timeout_duration = Duration::from_secs(1);
65            let timeout_future = tokio::time::sleep(timeout_duration);
66            tokio::pin!(timeout_future);
67
68            tokio::select! {
69                _ = &mut timeout_future => {
70                    info!("server task did not exit within {timeout_duration:?}, aborting server task");
71
72                    server_task_handle.abort();
73                    server_task_handle.await
74                }
75                result = &mut server_task_handle => result
76            }
77        }
78    };
79
80    let result = server_result
81        .context("failed to join server task")
82        .and_then(|r| r);
83
84    let shutdown_result = app_state
85        .shutdown()
86        .await
87        .context("failed to shutdown app state");
88
89    if let Err(error) = shutdown_result.as_ref() {
90        error!("{error}");
91    }
92
93    result.or(shutdown_result)
94}
