6/// Either the state if the channel lagged, or an update.
7#[derive(Debug)]
8pub enum StateUpdateItem<S>
9where
10    S: StateUpdateChannelState,
11{
12    /// The channel lagged.
13    State(S),
14
15    /// The update.
16    Update(S::Update),
17}
18
19/// Make a bounded channel.
20pub fn state_update_channel<S>(capacity: usize, state: S) -> (StateUpdateTx<S>, StateUpdateRx<S>)
21where
22    S: StateUpdateChannelState,
23{
24    let (tx, rx) = tokio::sync::broadcast::channel(capacity);
25
26    (
27        StateUpdateTx {
28            state: state.clone(),
29            stream: tx,
30        },
31        StateUpdateRx {
32            state,
33            stream: rx,
34            cloned: false,
35        },
36    )
37}

39/// A state that is changed via updates.
40pub trait StateUpdateChannelState: Clone {
41    /// The update that can be applied to this state.
42    type Update: Clone;
43
44    // TODO: Should this return an error if the update failed?
45    // If an error occurs while updating, this should be encoded into the state somehow?
46    //
47    /// Apply an update to this state.
48    ///
49    /// This function should be resilient to duplicate updates.
50    fn apply_update(&self, update: &Self::Update);
51}
52
53/// The sender for state updates.
54#[derive(Debug, Clone)]
55pub struct StateUpdateTx<S>
56where
57    S: StateUpdateChannelState,
58{
59    state: S,
60    stream: tokio::sync::broadcast::Sender<S::Update>,
61}
62
63impl<S> StateUpdateTx<S>
64where
65    S: StateUpdateChannelState,
66{
67    /// Send an update and apply it to the state.
68    pub fn send(&self, update: impl Into<S::Update>) {
69        let update = update.into();
70
71        self.state.apply_update(&update);
72
73        // TODO: How to handle no receivers?
74        let _ = self.stream.send(update).is_ok();
75    }
76
77    /// Get an immutable reference to the state.
78    ///
79    /// Use of this method is discouraged,
80    /// as state updates should typically be done through sending updates.
81    pub fn state_ref(&self) -> &S {
82        &self.state
83    }
84}

86/// The receiver for state updates
87#[derive(Debug)]
88pub struct StateUpdateRx<S>
89where
90    S: StateUpdateChannelState,
91{
92    state: S,
93    stream: tokio::sync::broadcast::Receiver<S::Update>,
94
95    cloned: bool,
96}
97
98impl<S> StateUpdateRx<S>
99where
100    S: StateUpdateChannelState,
101{
102    /// Get an immutable reference to the state.
103    ///
104    /// Use of this method is discouraged,
105    /// as state inspection should typically be done through listening on the channel.
106    pub fn state_ref(&self) -> &S {
107        &self.state
108    }
109
110    /// Get the next update in this stream, or the state if lagging occured.
111    ///
112    /// If an update occurs, the state should already be updated.
113    pub async fn recv(&mut self) -> Option<StateUpdateItem<S>> {
114        // If this rx handle was cloned, we might have lost messages.
115        // Re-send the state and clear the flag.
116        if self.cloned {
117            self.cloned = false;
118            return Some(StateUpdateItem::State(self.state.clone()));
119        }
120
121        let result = self.stream.recv().await;
122        match result {
123            Ok(update) => Some(StateUpdateItem::Update(update)),
124            Err(RecvError::Lagged(_n)) => Some(StateUpdateItem::State(self.state.clone())),
125            Err(RecvError::Closed) => None,
126        }
127    }
128}

155async fn make_stream_future<S>(
156    mut rx: StateUpdateRx<S>,
157) -> (Option<StateUpdateItem<S>>, StateUpdateRx<S>)
158where
159    S: StateUpdateChannelState,
160{
161    let item = rx.recv().await;
162    (item, rx)
163}

165type StreamFutureOutput<S> = (Option<StateUpdateItem<S>>, StateUpdateRx<S>);

167/// A Stream wrapper for a state update receiver
168#[derive(Debug)]
169pub struct StateUpdateStream<S>
170where
171    S: StateUpdateChannelState,
172{
173    future: ReusableBoxFuture<'static, StreamFutureOutput<S>>,
174}
175
176impl<S> Stream for StateUpdateStream<S>
177where
178    S: StateUpdateChannelState + Send + 'static,
179    S::Update: Send,
180{
181    type Item = StateUpdateItem<S>;
182
183    fn poll_next(
184        mut self: std::pin::Pin<&mut Self>,
185        cx: &mut core::task::Context<'_>,
186    ) -> std::task::Poll<Option<Self::Item>> {
187        let (result, rx) = ready!(self.future.poll(cx));
188        self.future.set(make_stream_future(rx));
189        std::task::Poll::Ready(result)
190    }
