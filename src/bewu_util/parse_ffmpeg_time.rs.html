4/// Parse an ffmpeg time, returning the time in seconds.
5///
6/// Format: `00:00:03.970612`, `HOURS:MM:SS.MICROSECONDS`
7pub fn parse_ffmpeg_time(time: &str) -> anyhow::Result<u64> {
8    // TODO: Find out why negative time can occur
9    let mut iter = time.trim_start_matches('-').split(':');
10
11    let hours = iter.next().context("missing hours")?;
12    let hours: u64 = hours
13        .parse()
14        .with_context(|| format!("failed to parse hours \"{hours}\""))?;
15    let minutes: u64 = iter
16        .next()
17        .context("missing minutes")?
18        .parse()
19        .context("failed to parse minutes")?;
20
21    let seconds = iter.next().context("missing seconds")?;
22    let (seconds, microseconds) = seconds
23        .split_once('.')
24        .context("failed to separate seconds and microseconds")?;
25    let _microseconds: u64 = microseconds.parse()?;
26    let seconds: u64 = seconds.parse()?;
27
28    ensure!(iter.next().is_none());
29
30    Ok((hours * 60 * 60) + (minutes * 60) + seconds)
31}
