1use lru::LruCache;
2use std::future::Future;
3use std::sync::Arc;
4use std::time::Duration;
5use std::time::Instant;
6
7/// An async LRU whose entries are valid for only a certian time period.
8///
9/// Entries must impl `Clone` as they are shared when racing and for the `valid_for` Duration.
10/// The simplest way to do this for a non-Clone type is an `Arc`.
11pub struct AsyncTimedLruCache<K, V> {
12    cache: std::sync::Mutex<LruCache<K, Arc<tokio::sync::OnceCell<Entry<V>>>>>,
13    valid_for: Duration,
14}
15
16impl<K, V> AsyncTimedLruCache<K, V>
17where
18    K: Eq + std::hash::Hash,
19    V: Clone,
20{
21    /// Make a new [`AsyncTimedLruCache`].
22    ///
23    /// # Panics
24    ///
25    /// Panics if the capacity is 0.
26    /// To simulate a 0-sized cache, instead use 0 for `valid_for` and a `capacity` of 1.
27    pub fn new(capacity: usize, valid_for: Duration) -> Self {
28        let capacity = capacity.try_into().expect("capacity is 0");
29        let cache = std::sync::Mutex::new(LruCache::new(capacity));
30        Self { cache, valid_for }
31    }
32
33    /// Get an entry by a value, providing a function that returns a future that is called if the cache entry is expired or empty.
34    ///
35    /// It is guaranteed that the function will be called only once if many gets run in parallel.
36    pub async fn get<FN, FUT>(&self, key: K, func: FN) -> V
37    where
38        FN: FnOnce() -> FUT,
39        FUT: Future<Output = V>,
40    {
41        let entry = {
42            let mut cache = self.cache.lock().expect("cache poisoned");
43
44            match cache.get_mut(&key) {
45                Some(entry) => match entry.get() {
46                    Some(cell_value) => {
47                        if cell_value.created.elapsed() > self.valid_for {
48                            std::mem::take(entry);
49                            entry.clone()
50                        } else {
51                            return cell_value.value.clone();
52                        }
53                    }
54                    None => {
55                        let entry = Arc::new(tokio::sync::OnceCell::new());
56                        cache.put(key, entry.clone());
57                        entry
58                    }
59                },
60                None => {
61                    let entry = Arc::new(tokio::sync::OnceCell::new());
62                    cache.put(key, entry.clone());
63                    entry
64                }
65            }
66        };
67
68        entry
69            .get_or_init(|| async move {
70                let value = func().await;
71                let created = Instant::now();
72
73                Entry { created, value }
74            })
75            .await
76            .value
77            .clone()
78    }
79}

81impl<K, V> std::fmt::Debug for AsyncTimedLruCache<K, V>
82where
83    K: std::hash::Hash + std::cmp::Eq,
84{
85    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
86        f.debug_struct("AsyncTimedLruCache")
87            .field("cache", &self.cache)
88            .field("valid_for", &self.valid_for)
89            .finish()
90    }
91}

93/// A cache entry
94#[derive(Debug)]
95struct Entry<V> {
96    /// The time this entry was created.
97    created: Instant,
98
99    /// The value of the entry.
100    value: V,
